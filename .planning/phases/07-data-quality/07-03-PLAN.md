**Step 1: Create scripts/prove-data-quality.js** — follows the exact same pattern as prove-attribution.js (mkdtempSync + env override + execSync node --test):

```js
'use strict';
const { execSync } = require('child_process');
const fs = require('fs');
const os = require('os');
const path = require('path');

const PROOF_DIR = path.join(__dirname, '..', 'proof');
const REPORT_JSON = path.join(PROOF_DIR, 'data-quality-report.json');
const REPORT_MD = path.join(PROOF_DIR, 'data-quality-report.md');

function run() {
  const tmpDir = fs.mkdtempSync(path.join(os.tmpdir(), 'rlhf-qual-proof-'));
  const env = { ...process.env, RLHF_FEEDBACK_DIR: tmpDir };

  const results = { passed: 0, failed: 0, requirements: {} };
  const checks = [
    {
      id: 'QUAL-01',
      desc: 'validate-feedback.js exports validateEntry with 4-level pipeline',
      fn: () => {
        delete require.cache[require.resolve('./validate-feedback')];
        const v = require('./validate-feedback');
        if (typeof v.validateEntry !== 'function') throw new Error('validateEntry not exported');
        if (typeof v.validateSchema !== 'function') throw new Error('validateSchema not exported');
        if (typeof v.validateSemantics !== 'function') throw new Error('validateSemantics not exported');
        if (typeof v.detectAnomalies !== 'function') throw new Error('detectAnomalies not exported');
        // Verify it catches semantic error
        const r = v.validateEntry({ id: 'x', timestamp: new Date().toISOString(), signal: 'positive', reward: -1, context: 'good work' });
        if (r.valid) throw new Error('Expected invalid entry with positive+negative-reward');
        if (!r.corrections.length) throw new Error('Expected auto-correction for reward');
      },
    },
    {
      id: 'QUAL-02',
      desc: 'captureFeedback produces richContext with domain, filePaths, errorType, outcomeCategory',
      fn: () => {
        process.env.RLHF_FEEDBACK_DIR = tmpDir;
        delete require.cache[require.resolve('./feedback-loop')];
        delete require.cache[require.resolve('./feedback-attribution')];
        const { captureFeedback } = require('./feedback-loop');
        const r = captureFeedback({ signal: 'positive', context: 'unit tests added', tags: ['testing'], filePaths: ['src/foo.js'] });
        if (!r.feedbackEvent) throw new Error('No feedbackEvent returned');
        const rc = r.feedbackEvent.richContext;
        if (!rc) throw new Error('richContext missing from feedbackEvent');
        if (!rc.domain) throw new Error('richContext.domain missing');
        if (!rc.outcomeCategory) throw new Error('richContext.outcomeCategory missing');
        if (!Array.isArray(rc.filePaths)) throw new Error('richContext.filePaths must be array');
      },
    },
    {
      id: 'QUAL-03',
      desc: 'inferOutcome returns granular categories beyond binary',
      fn: () => {
        delete require.cache[require.resolve('./feedback-loop')];
        const { inferOutcome } = require('./feedback-loop');
        if (typeof inferOutcome !== 'function') throw new Error('inferOutcome not exported');
        const cats = [
          inferOutcome('positive', 'solved it first try'),
          inferOutcome('positive', 'thorough comprehensive analysis'),
          inferOutcome('negative', 'wrong incorrect answer'),
          inferOutcome('negative', 'shallow surface response'),
          inferOutcome('negative', 'false assumption made'),
        ];
        const expected = ['quick-success', 'deep-success', 'factual-error', 'insufficient-depth', 'false-assumption'];
        cats.forEach((c, i) => {
          if (c !== expected[i]) throw new Error(`inferOutcome[${i}] expected ${expected[i]}, got ${c}`);
        });
      },
    },
    {
      id: 'QUAL-04',
      desc: 'test:quality passes with 0 failures',
      fn: () => {
        const out = execSync('node --test tests/validate-feedback.test.js', {
          cwd: path.join(__dirname, '..'),
          env: { ...process.env, RLHF_FEEDBACK_DIR: tmpDir },
          encoding: 'utf8',
        });
        if (out.includes('not ok') && !out.includes('# tests')) {
          throw new Error('Tests failed:\n' + out.slice(-500));
        }
      },
    },
  ];

  for (const check of checks) {
    try {
      check.fn();
      results.passed++;
      results.requirements[check.id] = { status: 'pass', desc: check.desc };
      console.log(`  PASS  ${check.id}: ${check.desc}`);
    } catch (err) {
      results.failed++;
      results.requirements[check.id] = { status: 'fail', desc: check.desc, error: err.message };
      console.error(`  FAIL  ${check.id}: ${err.message}`);
    }
  }

  fs.rmSync(tmpDir, { recursive: true, force: true });

  // Write proof artifacts
  fs.mkdirSync(PROOF_DIR, { recursive: true });
  const report = { phase: '07-data-quality', generatedAt: new Date().toISOString(), ...results };
  fs.writeFileSync(REPORT_JSON, JSON.stringify(report, null, 2) + '\n');

  const md = [
    '# Phase 7: Data Quality — Proof Report',
    '',
    `Generated: ${report.generatedAt}`,
    `Passed: ${results.passed} / ${checks.length}`,
    '',
    '## Requirements',
    '',
    ...Object.entries(results.requirements).map(([id, r]) =>
      `- [${r.status === 'pass' ? 'x' : ' '}] **${id}**: ${r.desc}${r.error ? `\n  - Error: ${r.error}` : ''}`
    ),
  ].join('\n');
  fs.writeFileSync(REPORT_MD, md + '\n');

  console.log(`\nPhase 7 proof: ${results.passed} passed, ${results.failed} failed`);
  if (results.failed > 0) process.exit(1);
}

run();
```
